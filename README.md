#aggregateTransform

A Node.js module for converting arrays of objects into aggregated, complex objects.

## Example use
```js
var t = require('aggregateTransform');

var template = {
  "{{accountNumber}}": {
    'revenueYearMonth': {
      '{{revenueYearMonth}}': {
        'utilities': {
          '{{utilityCode}}': {
            'sockets': {
              '{{serviceId}}': {
                'meters': {
                  '{{meter}}': {
                    'startDate': '{{startDate}}',
                    'endDate': '{{endDate}}',
                    'rate': '{{rate}}',
                    'utility': '{{utilityCode}}',
                    'account': '{{accountNumber}}',
                    'billPeriod': '{{revenueYearMonth}}',
                    'lineItems': {
                      '{{typeCode}}': {
                        'codes': {
                          '{{adjustmentCode}}': ['{{amount}}']
                        }
                      }
                    }
                  }
                }
              }
            }
          }
        }
      }
    }
  }
};

var obj = {};
var data = require('./input.js');

data.forEach(function (row) {
  t.transform(template, row, obj);
});

console.log(obj);
```
